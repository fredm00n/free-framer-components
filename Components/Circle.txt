// Circle function
export default function Circle(props) {
    // Stroke width
    let strokeWidth
    if (props.undo_stroke === true) {
        strokeWidth = 0
    } else {
        strokeWidth = props.strokeWidth
    }

    // Stroke color
    let oStrokeColor = Color(props.hslColorStroke)
    let strokeColor = fColor(
        oStrokeColor.h,
        props.rHueStroke,
        oStrokeColor.s * 100,
        props.rSaturationStroke,
        oStrokeColor.l * 100,
        props.rLuminosityStroke,
        oStrokeColor.a * 100,
        props.rAlphaStroke
    )

    // Define the random coordonates of circle points
    let points = fCircle(props.diameter, strokeWidth, props.intensity)

    // Return the code of circle
    let diameter = props.diameter // Diameter of circle

    // Random ID generator
    let randomID = generateString(6)
    let clipPathID = "mask_" + randomID
    let urlClipPathID = "url(#" + clipPathID + ")"

    // CSS styles
    const divStyles = {
        width: diameter,
        height: diameter,
        overflow: "visible",
    }
    const svgStyles = {
        width: diameter,
        height: diameter,
        overflow: "visible",
    }

    const childrenStyles = {
        clipPath: urlClipPathID,
    }

    // SVG circle return
    return (
        <div style={divStyles}>
            <svg style={svgStyles}>
                <clipPath id={clipPathID}>
                    <path d={points} fill="white" />
                </clipPath>
                <foreignObject
                    width={diameter}
                    height={diameter}
                    clipPath={urlClipPathID}
                >
                    <div style={childrenStyles}>{props.children}</div>
                </foreignObject>
                <path
                    d={points}
                    strokeWidth={strokeWidth}
                    stroke={strokeColor}
                    fill="none"
                />
            </svg>
        </div>
    )
}
